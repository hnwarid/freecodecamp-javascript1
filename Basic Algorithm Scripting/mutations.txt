function mutation(arr) {
  const test = arr[1].toLowerCase();
  const string = arr[0].toLowerCase();
  for (let i = 0; i < test.length; i++) {
    if (string.indexOf(test[i]) >= 0) {
      continue;
    } else
    return false;
  }
  return true;
}

console.log(mutation(["hello", "hey"]));
console.log(mutation(["Noel", "Ole"]));
